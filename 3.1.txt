Если ваш Python-сервер отправляет зашифрованные данные (AB 0C E6 43 D1), функция strstr в клиенте ищет строку 30 0D 0A 0D 0A ("0\r\n\r\n").     print(f"\n[GAME] {addr[0]}:{addr[1]} подключился к порту {port}")
    
    # 1. Формируем базовый хэндшейк
    base_handshake = b"0\r\n\r\n"

У нас классический Deadlock (взаимная блокировка). Сценарий такой:

Наш сервер отправляет зашифрованный хэндшейк (ab0ce643d1).
Клиент успешно его принимает, проверяет и устанавливает флаг param_1[0xf] = 2 (Подключение установлено).

Обе стороны ждут друг друга 10 секунд ➔ Таймаут ➔ Соединение рвется.

def handle_client(conn, addr, port):

import socket
import threading
import time

def handle_client(conn, addr, port):
    print(f"\n[GAME] {addr[0]}:{addr[1]} подключился к порту {port}")
    try:
        # 1. Пакет 0x7002 (Пустой ключ, для галочки)
        packet_seed = b'\x14\x00\x02\x70' + (b'\x00' * 16)

        # 2. Пакет 0x7000 (Рукопожатие / Старт)
        # В конце теперь стоят нули: 0x00 0x00 (Означает статус "ОК / Можно входить")
        packet_hello = b'\x06\x00\x00\x70\x00\x00'

        # Отправляем их вместе в открытом виде!
        start_packet = packet_seed + packet_hello
        conn.sendall(start_packet)
        print(f"[SENT] Отправлены Seed + Hello: {start_packet.hex()}")

        # Ждем ответ от клиента (Надеемся на логин и пароль!)
        conn.settimeout(15.0)
        data = conn.recv(2048)
        
        if data:
            print("\n" + "!"*50)
            print("БИНГО! ИГРА ОТВЕТИЛА В ОТКРЫТОМ ВИДЕ!!!")
            print(f"[RECV] Длина: {len(data)} байт")
            print(f"[RECV] Hex: {data.hex()}")
            try:
                # Пытаемся расшифровать как обычный текст
                print(f"[RECV] Текст: {data.decode('utf-8', errors='replace')}")
            except:
                print(f"[RECV] Текст: {repr(data)}")
            print("!"*50 + "\n")
        else:
            print("[GAME] Клиент закрыл соединение.")

    except socket.timeout:
        print(f"[TIMEOUT] Клиент молчит (таймаут 15с).")
    except Exception as e:
        print(f"[ERROR] Ошибка соединения: {e}")
    finally:
        conn.close()

def start_server(port):
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    try:
        s.bind(('0.0.0.0', port))
        s.listen(5)
        print(f"[*] Порт {port} слушает...")
        while True:
            conn, addr = s.accept()
            threading.Thread(target=handle_client, args=(conn, addr, port), daemon=True).start()
    except Exception as e:
        print(f"[!] Ошибка порта {port}: {e}")
