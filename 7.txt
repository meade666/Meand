Как написать свой сервер на Python
Основываясь на найденных структурах пакетов в command.h, мы можем написать скрипт-эмулятор сервера авторизации.

Создай файл kok3_login_server.py и запусти его. Затем попытайся зайти в клиент игры:

Python
import socket
import struct
import time
import random

# Константы из command.h
LOGON_USERCMD = 104
USER_REQUEST_LOGIN_PARA = 2
SERVER_RETURN_LOGIN_OK = 4
SERVER_RETURN_LOGIN_FAILED = 3

def handle_client(conn, addr):
    print(f"[*] Новое подключение от {addr}")
    try:
        while True:
            # 1. Читаем заголовок (допустим 2 байта размера + заголовок пакета)
            # В разных играх размер может быть 2 или 4 байта. 
            # Допустим, мы читаем первые сырые данные.
            data = conn.recv(1024)
            if not data:
                break
                
            print(f"[>] Получены данные ({len(data)} байт): {data.hex()}")
            
            # Проверим, вдруг это хендшейк 0x7002, о котором ты писал раньше
            if b"\x02\x70" in data or b"\x70\x02" in data:
                print("[!] Обнаружен хендшейк!")
                # Сервер должен что-то ответить, например, тот же хендшейк или ключ RSA
                # Если игра шлет RSA, нам надо будет его разобрать, но пока ответим просто эхом:
                conn.send(data)
                continue

            # Пытаемся распарсить как пакет логина (убирая заголовок размера, если он есть)
            # В command.h сказано: 
            # BYTE byCmd; BYTE byParam; DWORD dwTimestamp; ...
            # Попробуем найти команду 104 (0x68) и параметр 2 (0x02)
            
            idx = data.find(b'\x68\x02') # LOGON_USERCMD + USER_REQUEST_LOGIN_PARA
            if idx != -1:
                print("[!] Найден пакет запроса авторизации!")
                
                # Заголовок: BYTE byCmd(1), BYTE byParam(1), DWORD dwTimestamp(4) = 6 байт
                # Дальше идет:
                # char pstrName[48];
                # char pstrPassword[33];
                
                # Парсим логин и пароль
                login_start = idx + 6
                login_bytes = data[login_start:login_start+48]
                login = login_bytes.split(b'\x00')[0].decode('ascii', errors='ignore')
                
                pass_start = login_start + 48
                pass_bytes = data[pass_start:pass_start+33]
                password = pass_bytes.split(b'\x00')[0].decode('ascii', errors='ignore')
                
                print(f"[+] Логин: {login}")
                print(f"[+] Пароль: {password}")
                
                # Отправляем ответ SERVER_RETURN_LOGIN_OK (успешный логин)
                # struct stServerReturnLoginSuccessCmd:
                # BYTE byCmd (104), BYTE byParam (4), DWORD timestamp(4)
                # DWORD dwUserID(4), DWORD loginTempID(4)
                # char pstrIP[16], WORD wdPort(2)
                # BYTE key[256], DWORD state(4)
                
                byCmd = LOGON_USERCMD
                byParam = SERVER_RETURN_LOGIN_OK
                timestamp = int(time.time())
                user_id = 1
                temp_id = 12345
                ip = b"127.0.0.1".ljust(16, b'\x00')
                port = 8000  # Порт гейм сервера
                key = bytes([random.randint(0, 255) for _ in range(256)])
                state = 0
                
                # Пакуем пакет ответа (малоиндийский формат)
                response = struct.pack('<BBII I 16s H 256s I', 
                    byCmd, byParam, timestamp,
                    user_id, temp_id, ip, port, key, state
                )
                
                print(f"[<] Отправляю ответ об успешной авторизации ({len(response)} байт)")
                conn.send(response)
                
    except Exception as e:
        print(f"[-] Ошибка: {e}")
    finally:
        conn.close()
        print(f"[*] Соединение закрыто {addr}")

def main():
    host = '0.0.0.0'
    port = 1234  # Попробуй поменять на порт авторизации из клиента, если он другой

    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind((host, port))
    server.listen(5)
    print(f"[*] Сервер запущен на {host}:{port}")

    while True:
        conn, addr = server.accept()
        handle_client(conn, addr)

if __name__ == "__main__":
    main()

import http.server
import socketserver
import os

class MyHttpRequestHandler(http.server.SimpleHTTPRequestHandler):
    def do_GET(self):
        print(f"\n[HTTP] Клиент запросил: {self.path}")
        
        # Если клиент просит worldlist.xml в любом виде
        if "worldlist.xml" in self.path.lower():
            self.send_response(200)
            self.send_header("Content-type", "text/xml")
            self.end_headers()
            
            # Читаем наш файл worldlist.xml и отправляем клиенту
            with open("worldlist.xml", "rb") as f:
                self.wfile.write(f.read())
            print("[HTTP] Успешно отправлен worldlist.xml!")
            return
            
        # Если клиент просит patchfile.lst в любом виде
        elif "patchfile.lst" in self.path.lower():
            self.send_response(200)
            self.send_header("Content-type", "text/plain")
            self.end_headers()
            self.wfile.write(b"") # Отправляем пустоту
            print("[HTTP] Успешно отправлен пустой patchfile.lst!")
            return
            
        # Для всего остального
        self.send_response(404)
        self.end_headers()

# Настройки сервера
PORT = 80
handler_object = MyHttpRequestHandler

with socketserver.TCPServer(("", PORT), handler_object) as httpd:
    print(f"[*] Умный Web-сервер запущен на порту {PORT}")
    httpd.serve_forever()

import socket
import struct
import time
import random

# Константы команд из исходников игры (command.h)
LOGON_USERCMD = 104
USER_REQUEST_LOGIN_PARA = 2
SERVER_RETURN_LOGIN_OK = 4

def handle_client(conn, addr):
    print(f"\n[*] Новое подключение от {addr}")
    try:
        while True:
            data = conn.recv(1024)
            if not data:
                break
                
            print(f"[>] Получены данные ({len(data)} байт): {data.hex()}")
            
            # Проверка на хендшейк 0x7002
            if b"\x02\x70" in data or b"\x70\x02" in data:
                print("[!] Обнаружен хендшейк! Отправляем эхо...")
                # Пока просто отправляем те же байты обратно
                conn.send(data)
                continue

            # Ищем сигнатуру пакета логина: byCmd = 104 (0x68), byParam = 2 (0x02)
            idx = data.find(b'\x68\x02')
            if idx != -1:
                print("[!] Найден пакет запроса авторизации!")
                
                # Парсим логин (начинается через 6 байт после сигнатуры, длина 48)
                login_start = idx + 6
                login_bytes = data[login_start:login_start+48]
                login = login_bytes.split(b'\x00')[0].decode('ascii', errors='ignore')
                
                # Парсим пароль (сразу после логина, длина 33)
                pass_start = login_start + 48
                pass_bytes = data[pass_start:pass_start+33]
                password = pass_bytes.split(b'\x00')[0].decode('ascii', errors='ignore')
                
                print(f"[+] Логин: {login}")
                print(f"[+] Пароль: {password}")
                
                # --- Формируем ответ клиенту (Успешная авторизация) ---
                byCmd = LOGON_USERCMD
                byParam = SERVER_RETURN_LOGIN_OK
                timestamp = int(time.time())
                user_id = 1
                temp_id = 12345
                ip = b"127.0.0.1".ljust(16, b'\x00')
                port_game = 8000 # Порт следующего сервера (Game Server)
                key = bytes([random.randint(0, 255) for _ in range(256)])
                state = 0
                
                # Пакуем всё это в структуру C++ (малоиндийский формат)
                response = struct.pack('<BBII I 16s H 256s I', 
                    byCmd, byParam, timestamp,
                    user_id, temp_id, ip, port_game, key, state
                )
                
                print(f"[<] Отправляю ответ об успешной авторизации ({len(response)} байт)")
                conn.send(response)
                
    except Exception as e:
        print(f"[-] Ошибка: {e}")
    finally:
        conn.close()
        print(f"[*] Соединение закрыто {addr}")

def main():
    host = '0.0.0.0'
    port = 4704  # Тот самый порт, который мы указали в worldlist.xml

    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # Позволяет быстро перезапускать сервер без ошибки "Порт занят"
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) 
    server.bind((host, port))
    server.listen(5)
    
    print(f"[*] TCP Сервер авторизации запущен на {host}:{port}")
    print("[*] Ожидание подключения игры...")

    while True:
        try:
            conn, addr = server.accept()
            handle_client(conn, addr)
        except KeyboardInterrupt:
            print("\n[*] Сервер остановлен.")
            break

if __name__ == "__main__":
    main()

    byParam = 4       # SERVER_RETURN_LOGIN_OK

Почему клиент молчит, пока не закроешь?
В старых китайских MMO (и в исходниках Giant Engine это часто встречается) алгоритм входа выглядит так:

Клиент подключается по TCP.
Сервер отправляет клиенту размер ключа и сам ключ шифрования (чаще всего RSA-ключ или Seed для XOR-шифровки).
Клиент генерирует логин/пароль, шифрует их этим ключом и отправляет серверу (те самые 71 байт).
Сервер проверяет и пускает.
То есть клиент не может отправить логин первым, пока сервер не даст ему "добро" и ключ шифрования. Когда ты закрываешь клиент, он экстренно завершает все сетевые потоки и, видимо, сбрасывает буфер, где уже лежал "недособранный" или зашифрованный "нулевым" ключом пакет логина, плюс пакет quit.

Мы знаем из исходников, что есть команда USER_VERIFY_VER_PARA = 120. Сервер часто требует проверки версии перед входом. Но, скорее всего, сервер должен послать пакет stServerReturnLoginSuccessCmd не после логина, а какие-то данные для шифрования.

import socket
import threading
import time
import struct

def build_login_success_packet():
    """Собирает пакет успешной авторизации с правильным заголовком размера"""
    byCmd = 104       # LOGON_USERCMD
    byParam = 4       # SERVER_RETURN_LOGIN_OK
    timestamp = int(time.time()) & 0xFFFFFFFF
    user_id = 777     # Твой уникальный ID
    temp_id = 12345   # ID сессии
    ip = b"127.0.0.1".ljust(16, b'\x00') # IP следующего (Game) сервера
    port = 8000       # Порт Game-сервера
    key = bytes([0] * 256) # Пустой ключ шифрования сессии
    state = 1         # Статус (1 = нормальный)

    # Упаковываем саму структуру пакета (292 байта)
    payload = struct.pack('<BBIII16sH256sI', 
        byCmd, byParam, timestamp,
        user_id, temp_id, ip, port, key, state
    )

    packet_size = len(payload) + 2

    return size_header + payload

    """Собирает идеальный пакет успешного входа"""
    byCmd = 104       # LOGON_USERCMD
    byParam = 4       # SERVER_RETURN_LOGIN_OK
    timestamp = int(time.time()) & 0xFFFFFFFF
    user_id = 777     # Уникальный ID аккаунта
    temp_id = 12345   # ID сессии
    ip = b"127.0.0.1".ljust(16, b'\x00') # IP Game-сервера
    port = 8000       # Порт Game-сервера
    key = bytes([0] * 256) # Ключ сессии
    state = 1         # Статус аккаунта (1 - норм)

    # Упаковываем внутренности (Little Endian)
    payload = struct.pack('<BBIII16sH256sI', 
        byCmd, byParam, timestamp,
        user_id, temp_id, ip, port, key, state
    )

    # Итоговый пакет: 01 26 (294 байта) + payload
    return size_header + payload

attempt_counter = 0

def get_next_handshake():
    global attempt_counter
    attempt = attempt_counter % 5
    attempt_counter += 1
    
    if attempt == 0:
        # Вариант 1: Команда 0x70 (Handshake), размер в Big Endian
        print("[!] Пробуем вариант 1 (Handshake, Big Endian size)")
        return struct.pack('>H', 8) + b"\x70\x02\x11\x22\x33\x44"
    elif attempt == 1:
        # Вариант 2: Команда 0x70, размер в Little Endian
        print("[!] Пробуем вариант 2 (Handshake, Little Endian size)")
        return struct.pack('<H', 8) + b"\x70\x02\x11\x22\x33\x44"
    elif attempt == 2:
        # Вариант 3: Пинг (Cmd=2), Little Endian
        print("[!] Пробуем вариант 3 (Ping)")
        return struct.pack('<H BB H', 6, 2, 2, 0)
    elif attempt == 3:
        # Вариант 4: Проверка версии (Cmd=104, Param=120)
        print("[!] Пробуем вариант 4 (Проверка версии)")
        payload = struct.pack('<BBIII', 104, 120, 0, 0, 1999)
        return struct.pack('<H', len(payload) + 2) + payload
    elif attempt == 4:
        # Вариант 5: Пустой пакет (просто размер 0)
        print("[!] Пробуем вариант 5 (Пустой пакет)")
        return struct.pack('<H', 0)

import socket

import struct

def listen_port(port):
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    # Отключаем буферизацию пакетов, чтобы они улетали сразу
    server.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
    
    try:
        server.bind(('0.0.0.0', port))
        server.listen(5)
    except Exception:
        return

    while True:
        try:
            client_sock, address = server.accept()
            client_sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
            print(f"\n[+] ИГРА ПОДКЛЮЧИЛАСЬ! Порт {port}")
            threading.Thread(target=fuzz_client, args=(client_sock, port), daemon=True).start()
        except Exception:
            pass

def fuzz_client(client_sock, port):
    client_sock.settimeout(0.5) # Очень короткий таймаут для чтения
    
    # Мы будем перебирать все возможные комбинации Cmd (от 1 до 10) и Param (от 1 до 10)
    # Плюс стандартные из исходников (104, 2, 0x70)
    test_commands = [
        (0x70, 0x02), # Handshake
        (104, 120),   # Verify Version
        (104, 4),     # Login Success
        (2, 2),       # Game Time
        (2, 1),       # Request Time
        (3, 53)       # Merge Version
    ]
    
    # Добавим просто перебор всех байт от 1 до 20
    for cmd in range(1, 20):
        for param in range(1, 5):
            test_commands.append((cmd, param))
            
    print(f"[*] Начинаем FUZZING! Будем отправлять разные пакеты каждую секунду...")
    
    try:
        for cmd, param in test_commands:
            # Создаем тестовый пакет (Cmd, Param, Timestamp)
            payload = struct.pack('<BB I', cmd, param, int(time.time()))
            
            # В Giant Engine размер включает в себя и сам заголовок
            size = len(payload) + 2
            
            # Пробуем отправить с Little Endian размером
            packet_le = struct.pack('<H', size) + payload
            
            try:
                client_sock.send(packet_le)
                print(f"[>] Отправлено Cmd:{cmd} Param:{param}")
            except:
                print("[-] Клиент разорвал связь.")
                return

            # Слушаем полсекунды
            try:
                data = client_sock.recv(4096)
                if data:
