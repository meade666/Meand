from http.server import HTTPServer, BaseHTTPRequestHandler
import socket
import threading
import time

WORLDLIST_XML = """<?xml version="1.0" encoding="utf-16"?>
<server>
    <zones>
        <zone id="0" name="RU" type="0">
            <world id="0" name="WE" port="4981" status="1" type="0">192.168.1.184</world>
        </zone>
    </zones>
</server>"""

WORLDLIST = b'\xff\xfe' + WORLDLIST_XML.encode('utf-16-le')
PATCHFILE = b'\xff\xfe' + "0\r\n".encode('utf-16-le')

class Handler(BaseHTTPRequestHandler):
    def do_GET(self):
        path = self.path
        host = self.headers.get('Host')
        user_agent = self.headers.get('User-Agent')
        
        print(f"\n[HTTP GET] {path}")
        print(f"[HOST] {host}")
        print(f"[USER-AGENT] {user_agent}")

        if 'worldlist.xml' in path:
            self.send_response(200)
            self.send_header('Content-Type', 'text/xml; charset=utf-16')
            self.send_header('Content-Length', str(len(WORLDLIST)))
            self.end_headers()
            self.wfile.write(WORLDLIST)
            print("[SENT] worldlist.xml")
            
        elif 'patchfile.lst' in path:
            self.send_response(200)
            self.send_header('Content-Type', 'text/plain')
            self.send_header('Content-Length', str(len(PATCHFILE)))
            self.end_headers()
            self.wfile.write(PATCHFILE)
            print("[SENT] patchfile.lst")
            
        else:
            print(f"[UNKNOWN] {path}")
            self.send_response(200)
            self.send_header('Content-Type', 'text/plain')
            self.end_headers()
            self.wfile.write(b"OK")
            print("[SENT] OK")

    def log_message(self, format, *args):
        pass

def start_http_server():
    print("[*] HTTP Сервер запущен на порту 80...")
    server = HTTPServer(('0.0.0.0', 80), Handler)
    server.serve_forever()

def xor_encrypt(data, key):
    key_bytes = bytes.fromhex(key)
    result = bytearray()
    for i, byte in enumerate(data):
        result.append(byte ^ key_bytes[i % len(key_bytes)])
    return bytes(result)

def handle_game_client(conn, addr, port):
    print(f"\n[GAME] {addr[0]}:{addr[1]} подключился к порту {port}")
    
    handshake = b"0\r\n\r\n"
    encrypted = xor_encrypt(handshake, "BB40E64E")
    
    conn.send(encrypted)
    print(f"[SENT] Handshake: {encrypted.hex()}")
    
    conn.settimeout(10)
    
    try:
        data = conn.recv(4096)
        if data:
            print(f"[RECV] {len(data)} bytes: {data.hex()}")
        else:
            print("[RECV] Пусто")
    except socket.timeout:
        print("[TIMEOUT]")
    except Exception as e:
        print(f"[ERROR] {e}")
    finally:
        conn.close()

def listen_port(port):
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    s.bind(('0.0.0.0', port))
    s.listen(5)
    print(f"[*] Порт {port} слушает...")
    
    while True:
        try:
            conn, addr = s.accept()
            threading.Thread(target=handle_game_client, args=(conn, addr, port), daemon=True).start()
        except Exception as e:
            print(f"[ERROR] Порт {port}: {e}")

if __name__ == '__main__':
