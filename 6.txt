        if (ins == null) return false;
        if (string.IsNullOrEmpty(json)) return false;
        if (file_uuid < 0) return false;
        if (cb == null) return false;

    public bool AddUploadTask(string json, VoidFuncN<bool> cb, long file_uuid, GameObject host)
    {
        if (string.IsNullOrEmpty(json)) return false;
        if (isWorking) return false;
        if (cb == null) return false;
        if (this.cb != null) return false;
        if (file_uuid < 0) return false;
        if (host == null) return false;
        isWorking = true;
        this.cb = cb;
        this.file_uuid = file_uuid;
        StartCoroutine(RunUploadTask(json));
        return true;
    }
    IEnumerator RunUploadTask(string json)
    {
        WWWForm form = new WWWForm();
        form.AddField("json_data", json);
        form.AddField("file_name", file_uuid.ToString());

        WWW www = new WWW(upload_url, form);

        yield return www;
        if (www.text == "ok")
        {
            if (host != null)
            {
                cb.Invoke(true);
            }
        }
        else if (www.text == "error")
        {
            if (host != null)
            {
                cb.Invoke(false);
            }
        }
        else
        {
            // upload php error
            Debug.LogError(www.text);
            if (host != null)
            {
                cb.Invoke(false);
            }
        }
#if UNITY_EDITOR
        Debug.Log("upload map response:" + www.text);
#endif
        cb = null;
        isWorking = false;
    }

public static IEnumerator Download(string name, VoidFuncString ok, 
                                   VoidFuncVoid error = null)
{
    // 1. Сначала пытается загрузить из локального кэша
    string json = Serializable.LoadFromFileEx(name + ".json");
    
    if (string.IsNullOrEmpty(json))
    {
        // 2. Если нет - скачивает с сервера (3 попытки)
        int try_times = 3;
        while (try_times > 0)
        {
            string url = Serializable.GetDownloadWWWUrl(name);
            var www = new WWW(url);
            yield return www;
            
            // 3. Проверяет что это валидный JSON
            if (Serializable.Map.IsMapJson(www.text))
            {
                // 4. Сохраняет в локальный кэш
                Serializable.SaveToFile(name + ".json", www.text);
                json = www.text;
                break;
            }
            else
            {
                if (www.text.IndexOf("Document not found") > 0)
                {
                    Debug.LogError("This Map DoesNot exist " + name);
                    try_times = -1;
                    break;
                }
                try_times--;
            }
        }
    }
    
    // 5. Вызывает callback
    if (ok != null)
        ok(json);
}

public void Load(long customs = 0)
{
    // 1. Загружает файл карты по ID
    string file = customs.ToString() + ".json";
    var str = Serializable.LoadFromFile(file);
    
    // 2. Десериализует JSON в объект Map
    Serializable.Map map = Serializable.ToObject<Serializable.Map>(str);
    
    // 3. Получает все компоненты EditorSerializeBase в детях
    var serializes = this.transform.GetComponentsInChildren<EditorSerializeBase>();
    
    // 4. Для каждого компонента десериализует объект
    foreach (var p in serializes)
    {
        p.DeSerializeObject(map);  // ← ГЛАВНОЕ: восстанавливает объекты из JSON
    }
    
    // 5. Загружает фон по теме
    LoadBackGround(map.theme);
}

// C# ASP.NET Core
public class MapController : ControllerBase
{
    [HttpPost("upload.php")]
    public ActionResult Upload([FromForm] string json_data, 
                               [FromForm] string file_name)
    {
        // 1. Сохранить файл
        var mapPath = Path.Combine(_mapsDir, file_name + ".json");
        File.WriteAllText(mapPath, json_data);
        
        // 2. Парсить JSON
        var map = JsonConvert.DeserializeObject<Serializable.Map>(json_data);
        
        // 3. Сохранить в БД
        _dbContext.Maps.Add(new MapModel 
        { 
            Id = long.Parse(file_name),
            JsonData = json_data,
            Theme = map.theme
        });
        await _dbContext.SaveChangesAsync();
        
        return Ok("ok");
    }
    
    [HttpGet("maps/{id}")]
    public ActionResult Download(long id)
    {
        var map = await _dbContext.Maps.FindAsync(id);
        if (map == null) 
            return NotFound("Document not found");
        
        return Ok(map.JsonData);
    }
}

// C# .NET Core
public class GameServer
{
    private TcpListener _listener;
    private Dictionary<long, GameSession> _sessions = new();
    
    public async Task Start(string host, int port)
    {
        _listener = new TcpListener(IPAddress.Parse(host), port);
        _listener.Start();
        
        while (true)
        {
            var client = await _listener.AcceptTcpClientAsync();
            _ = HandleClientAsync(client);
        }
    }
    
    private async Task HandleClientAsync(TcpClient client)
    {
        try
        {
            var stream = client.GetStream();
            var session = new GameSession(client, stream, _sessions);
            
            // 1. Регистрируем обработчики (как в Go)
            session.RegisterHandler(1001, HandleLogin);
            session.RegisterHandler(1006, HandleMove);
            
            // 2. Начинаем читать сообщения
            await session.StartAsync();
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Error: {ex.Message}");
        }
        finally
        {
            client.Close();
        }
    }
    
    private async Task<bool> HandleLogin(GameSession session, byte[] data)
    {
        // Парсим логин
        var username = Encoding.UTF8.GetString(data);
        
        // Проверяем в БД
        var user = await _dbContext.Users.FirstOrDefaultAsync(u => u.Username == username);
        if (user == null)
            return false;
        
        session.UserId = user.Id;
        session.Username = username;
        
        // Отправляем ответ
        await session.SendAsync(1001, Encoding.UTF8.GetBytes("ok"));
        
        return true;
    }
    
    private async Task<bool> HandleMove(GameSession session, byte[] data)
    {
        // Парсим позицию (X, Y, Z)
        var pos = ParsePosition(data);
        
        // Обновляем в БД
        var player = await _dbContext.Players.FindAsync(session.UserId);
        player.X = pos.X;
        player.Y = pos.Y;
        player.Z = pos.Z;
        await _dbContext.SaveChangesAsync();
        
        // Отправляем всем игрокам на карте
        await BroadcastToMapAsync(player.MapId, 1006, data);
        
        return true;
    }
}
