def listen_port(port):
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    try:
        server.bind(('0.0.0.0', port))
        server.listen(5)
        print(f"[*] Порт {port} слушает...")
    except Exception as e:
        print(f"[!] Не могу открыть порт {port}: {e}")
        return

    while True:
        try:
            client_sock, address = server.accept()
            print(f"\n[+] Игра подключилась на порт {port} с адреса {address[0]}:{address[1]}")
            threading.Thread(
                target=handle_client,
                args=(client_sock, port),
                daemon=True
            ).start()
        except Exception as e:
            print(f"[!] Ошибка принятия подключения на порту {port}: {e}")
