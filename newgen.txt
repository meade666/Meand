PORTS = [4700, 4701, 4702, 4703, 4704, 4981, 4983, 4985, 4987]

def make_pkt(opcode, size):
    body = b'\x00' * size
    pkt_size = 2 + 2 + len(body)
    return struct.pack('!HH', pkt_size, 1) + struct.pack('!H', opcode) + body

def handle(conn, addr):

    print(f"[+] Порт {port} от {addr}")

    t = int(time.time())
    body81 = struct.pack('!hxxIIII', 1, t, t, t, 0)
    pkt81_size = 2 + 2 + len(body81)
    pkt81 = struct.pack('!HH', pkt81_size, 1) + struct.pack('!H', 0x81) + body81
    conn.sendall(pkt81)
    print(f"[>] Пакет 0x81 отправлен")
    time.sleep(0.1)

    # (opcode, размер тела из FUN_006bbd00)

        (1, 4),      # size=4
        (4, 0xc),    # size=12
        (0x7001, 2), # size=2
        (0x7002, 0x10), # size=16
        (0x80, 8),   # size=8
        (0x82, 8),   # size=8
    ]

        conn.sendall(pkt)

HOST = '0.0.0.0'
PORTS = [4700, 4701, 4702, 4703, 4704, 4981, 4983, 4985, 4987]

def handle(conn, addr):
    port = conn.getsockname()[1]
    print(f"[+] Порт {port} от {addr}")
    
    # Version Tag — 8 байт без паддинга
    version_tag = struct.pack('!HHI', 8, 0x1200, 0x00030000)
    conn.sendall(version_tag)
    print(f"[>] Version Tag: {version_tag.hex()}")
    time.sleep(0.1)
    
    # Пакет 0x81 — без паддинга, только нужные данные
    # Внутри: opcode(2) + case(2) + 4xuint32(16) = 20 байт
    t = int(time.time())
    inner = struct.pack('!HhIIII', 0x81, 1, t, t, t, 0)
    # Заголовок Version Tag для этого пакета
    pkt81 = struct.pack('!HHI', 8 + len(inner), 0x1200, 0x00030000) + inner
    conn.sendall(pkt81)
    print(f"[>] Пакет 0x81: {pkt81.hex()}")
    
    conn.settimeout(60)
    try:
        while True:
            data = conn.recv(65536)
            if not data:
                print("[DISCONNECTED]")
                break
            print(f"[<] {len(data)} байт: {data.hex()}")
    except socket.timeout:
        print("[TIMEOUT]")
    conn.close()

def listen(port):
    s = socket.socket()
    s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    s.bind((HOST, port))
    s.listen(5)
    while True:
        conn, addr = s.accept()
        threading.Thread(target=handle, args=(conn, addr)).start()

for port in PORTS:
    threading.Thread(target=listen, args=(port,)).start()
